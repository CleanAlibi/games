<body>

  <div id="instructions">Use arrow keys to adjust speed. Slow down when sewing over pins!</div>

  <canvas id="gameCanvas" width="600" height="400"></canvas>

  <script>

    const canvas = document.getElementById('gameCanvas');

    const ctx = canvas.getContext('2d');


    let gameSpeed = 2;

    let y = 200;

    let time = 0;

    let score = 0;

    let needleBroken = false;

    let pinLocations = [];

    const pinFrequency = 4000; // milliseconds


    function drawMachineView() {

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      

      // Sewing guide line

      ctx.beginPath();

      ctx.moveTo(0, y);

      ctx.lineTo(canvas.width, y);

      ctx.strokeStyle = '#000';

      ctx.lineWidth = 2;

      ctx.stroke();


      // Draw needle

      ctx.beginPath();

      ctx.moveTo(300, y - 20);

      ctx.lineTo(300, y + 20);

      ctx.strokeStyle = needleBroken ? 'red' : 'blue';

      ctx.lineWidth = 4;

      ctx.stroke();


      // Draw pins

      pinLocations.forEach(pinX => {

        ctx.beginPath();

        ctx.moveTo(pinX, y - 10);

        ctx.lineTo(pinX, y + 10);

        ctx.strokeStyle = 'green';

        ctx.lineWidth = 2;

        ctx.stroke();

      });


      // Score

      ctx.fillStyle = '#333';

      ctx.font = '16px sans-serif';

      ctx.fillText('Score: ' + score, 10, 20);

    }


    function updatePins() {

      for (let i = 0; i < pinLocations.length; i++) {

        pinLocations[i] -= gameSpeed;

        // If the pin reaches the needle

        if (pinLocations[i] < 304 && pinLocations[i] > 296) {

          if (gameSpeed > 3) {

            needleBroken = true;

            return;

          } else {

            score += 10;

          }

        }

      }

      // Remove off-screen pins

      pinLocations = pinLocations.filter(x => x > 0);

    }


    function gameLoop() {

      if (needleBroken) return;

      time += 16;

      if (time % pinFrequency < 16) {

        pinLocations.push(canvas.width);

      }

      updatePins();

      drawMachineView();

      gameSpeed += 0.001; // Slowly increase speed

      requestAnimationFrame(gameLoop);

    }


    document.addEventListener('keydown', e => {

      if (e.key === 'ArrowDown') {

        gameSpeed = Math.max(1, gameSpeed - 1);

      } else if (e.key === 'ArrowUp') {

        gameSpeed = Math.min(10, gameSpeed + 1);

      } else if (e.key === 'r' && needleBroken) {

        // Restart game

        gameSpeed = 2;

        needleBroken = false;

        pinLocations = [];

        score = 0;

        time = 0;

        gameLoop();

      }

    });


    drawMachineView();

    gameLoop();

  </script>
